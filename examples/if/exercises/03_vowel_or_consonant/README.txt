//  ./main.exe
//   Give me a letter.
//
//  ./main.exe hi
//   Give me a letter.
//
//  ./main.exe e
//   "e" is a vowel.
//
//  ./main.exe p
//   "p" is a consonant.

//  ./main.exe y
//   "y" is sometimes a vowel, sometimes not.
